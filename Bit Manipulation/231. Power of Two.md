- if n is possitive and theres only one 1 bit in its binary representation then the number is a power of 2. :)

# Complexity
- Time complexity:
O(1)

- Space complexity:
O(1)

# Code
```cpp []
class Solution {
public:
    bool isPowerOfTwo(int n) {
     if(n<=0) return false; //important
     int c = 0;
     for(int i=0; i<31; i++){ //no need to go for the last bit as negative value is not power of 2
        if(n&1){
            c++;
            if(c>1) return false;
        }
        n >>=1;
     }   
     return true;
    }
};
```
