
// ðŸ”¹ What is __builtin_popcount()?

// It is a GCC/Clang compiler built-in function that counts the number of set bits (1s) in a number.

// Popcount = Population Count
// â†’ Number of 1s in binary representation.

Time = O(1)
Space = O(1)   (excluding output)

# Code
```cpp []


class Solution {
public:
    vector<string> readBinaryWatch(int turnedOn) {
        vector<string> ans;
        for(int i=0; i<12; i++){
            for(int j=0; j<60; j++){
                if(__builtin_popcount(i) + __builtin_popcount(j) == turnedOn){
                    ans.push_back(to_string(i) + ":" + (j<10? "0":"") + to_string(j));
                }
            }
        }

        return ans;
    }
};

```

 class Solution {
 public:
     void no_of_ones(vector<int>& v, int l){
         for(int i=0; i<l; i++){
             int ones = 0;
             int n=i;
             while(n){
                 if(n&1) ones++;
                 n >>= 1;
             }
             v[i] = ones;
         }
     }
     vector<string> readBinaryWatch(int turnedOn) {
         vector<int> hour(12,0);
         no_of_ones(hour, 12);
         vector<int> minutes(60,0);
         no_of_ones(minutes, 60);

         vector<string> ans;
         for(int i=0; i<12; i++){
            
             for(int j=0; j<60; j++){
                 if(hour[i]+minutes[j] == turnedOn){
                     ans.push_back(to_string(i) + ":" + (j<10? "0":"") + to_string(j));
                 }

             }
         }

         return ans;

     }
 };
