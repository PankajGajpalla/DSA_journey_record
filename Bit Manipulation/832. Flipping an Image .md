

# Complexity
- Time complexity:
O(m * n) , where m = no.of rows, n = no. of columns

- Space complexity:
O(1)

# Code
```cpp []
class Solution {
public:
    vector<vector<int>> flipAndInvertImage(vector<vector<int>>& image) {

        for(auto &c: image){
            int i = 0;
            int j = c.size()-1;
            while(i<=j){
                if(c[i] == c[j]){
                    c[i] ^= 1;
                    c[j] = c[i];
                }
                i++;
                j--;                
            }
        }

        return image;
    }
};
```
