time complexity = O(1)
space  =  O(1)

# Code
```cpp []
class Solution {
public:
    int hammingDistance(int x, int y) {
        int n = x^y;
        int i = 0;
        while(n){
            i += n&1;
            n>>=1; 
        }
        return i;
    }
};
```
