## time complexity - O(8) = O(1)
## space complexity - O(1)

# Code
```cpp []
class Solution {
public:
    string toHex(int num) {
        if(num==0) return "0";

        unsigned int n = num;

        string hex = "0123456789abcdef";
        
        string ans = "";
        while(n){
            ans = hex[n&15] + ans;
            n >>= 4;
        }

        return ans;
    }
};
```
