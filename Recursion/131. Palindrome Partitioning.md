class Solution:
    def partition(self, s: str) -> List[List[str]]:
        result = []
        
        def dfs(start, part):
            if start>=len(s):
                result.append(part.copy())
                return
            
            for i in range(start, len(s)):
                if self.pali(s,start, i):
                    part.append(s[start:i+1])
                    dfs(i+1, part)
                    part.pop()

        dfs(0, [])
        return result
    
    def pali(self,s, l, r):
        # if not st: return False
        # l,r = 0, len(st)-1
        while(l<r):
            if s[l] != s[r]:
                return False
            l,r = l+1, r-1
        
        return True
